# Activate the venv first
source venv/bin/activate

# Basic usage — run all 5 stages, 2-minute scan
python main.py -n "site-a" -l "Building 3, Floor 2"

# Stage 1 only (passive map, 30s scan)
python main.py --stages 1 --scan-duration 30 -n test

# Stages 1 and 2 only (map + connection intel)
python main.py --stages 1,2 -n test

# Skip the confirmation prompts on active stages (3/4/5)
python main.py --stages 1,2,3,5 --no-gate -n test

# Focus on a specific device
python main.py --stages 1,5 --target AA:BB:CC:DD:EE:FF -n test

# Resume an existing engagement (adds to same DB record)
python main.py --engagement abc123def456 -n test

# Training mode — see every WHAD API call narrated
# Edit config.py: VERBOSE_MODE = True
python main.py --stages 1 --scan-duration 30 -n training
Key flags:

Flag	Default	Description
-n / --name	unnamed	Engagement name
-l / --location	empty	Physical location
-e / --engagement	auto-generated	Resume an existing engagement ID
--stages	1,2,3,4,5	Which stages to run
--scan-duration	120	Stage 1 scan length (seconds)
--no-gate	off	Skip active-stage confirmation prompts
--target	none	Filter to one BD address (repeatable)
--interface	uart0	WHAD interface
Outputs land in:

findings.db — SQLite, all engagements
reports/report_<id>.md — Markdown
reports/report_<id>.json — JSON
pcaps/ — packet captures from S2
